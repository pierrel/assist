   524	    """Merge the current branch into main with AI-generated summary."""
   525	    try:
   526	        thread = MANAGER.get(tid)
   527	    except FileNotFoundError:
   528	        raise HTTPException(status_code=404, detail="Thread not found")
   529	
   530	    dm = _get_domain_manager(tid)
   531	    if not dm or not dm.repo:
   532	        raise HTTPException(status_code=400, detail="No git repository configured for this thread")
   533	
   534	    # Get a model for summarizing
   535	    from assist.model_manager import select_chat_model
   536	    try:
   537	        summary_model = select_chat_model("gpt-oss-20b", temperature=0.1)
   538	    except Exception:
   539	        # If model fails to load, pass None and use fallback summary
   540	        summary_model = None
   541	
   542	    try:
   543	        summary = dm.merge_to_main(summary_model)
   544	        # Redirect with success message (could use query param)

   546	            url=f"/thread/{tid}?merged=1",
   547	            status_code=303

   550	        # User-friendly error (merge conflicts, no changes, etc.)
   551	        raise HTTPException(status_code=400, detail=str(e))
   552	    except subprocess.CalledProcessError as e:
   553	        # Git command failed
   554	        raise HTTPException(status_code=500, detail=f"Git operation failed: {e}")
   555	    except Exception as e:
   556	        # Unexpected error
   557	        logging.error(f"Merge failed for thread {tid}: {e}", exc_info=True)
   558	        raise HTTPException(status_code=500, detail=f"Merge failed: {str(e)}")
