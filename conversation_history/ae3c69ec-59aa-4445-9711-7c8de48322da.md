   141	def render_diff(text: str) -> str:
   142	    """Render a diff using Pygments.
   143	    
   144	    If the diff is too large, truncate it to avoid performance issues.
   145	    The truncation limit is controlled by the `DIFF_TRUNCATE_LIMIT` environment variable.
   146	    Defaults to 1000 lines if not set.
   147	    
   148	    Truncation occurs at the end of a hunk (lines starting with '@@') to ensure
   149	    logical completeness.
   150	    """
   151	    # Get truncation limit from environment variable or use default
   152	    truncate_limit = int(os.getenv("DIFF_TRUNCATE_LIMIT", "1000"))
   153	    
   154	    # Truncate the diff if it exceeds the limit
   155	    lines = text.splitlines()
   156	    if len(lines) > truncate_limit:
   157	        # Find the last complete hunk (lines starting with '@@') before the truncate limit
   158	        last_hunk_index = -1
   159	        for i in range(min(truncate_limit, len(lines)) - 1, -1, -1):
   160	            if lines[i].startswith('@@'):
   161	                last_hunk_index = i
   162	                break
   163	        
   164	        # If no hunk found, just truncate to the limit
   165	        if last_hunk_index == -1:
   166	            truncated_lines = lines[:truncate_limit]
   167	        else:
   168	            truncated_lines = lines[:last_hunk_index + 1]
   169	        
   170	        truncated_text = "\n".join(truncated_lines)

   743	    from manage.eval_history import get_runs
   744	
   745	    runs = get_runs(limit=50)

   748	        return f"<html><body><p>Run '{html.escape(run_id)}' not found.</p><a href='/evals'>Back</a></body></html>"
   749	
   750	    result = run["tests"].get(test_key)

   752	        return (
   753	            f"<html><body><p>Test not found in run {html.escape(run_id)}.</p>"
   754	            f"<a href='/evals'>Back</a></body></html>"
